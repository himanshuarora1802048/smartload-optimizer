MAKEFILE COMMANDS (NEW!)

Build & Test:
  make build              # Build JAR with Maven
  make test               # Run all tests
  make clean              # Clean build artifacts

Run Locally:
  make run                # Run application (foreground)
  make run-bg             # Run in background
  make stop               # Stop background app

Docker:
  make docker-build       # Build Docker image
  make docker-run         # Run in Docker container
  make docker-stop        # Stop Docker container
  make docker-logs        # View container logs

Docker Compose:
  make compose-up         # Start with compose
  make compose-down       # Stop compose services
  make compose-logs       # View compose logs

Testing:
  make test-api           # Run integration tests
  make test-sample        # Test with sample data
  make test-performance   # Test with 22 orders
  make health-check       # Check service health

Quick Start:
  make help               # Show all commands
  make all                # Clean, build, docker-build
  make quick-start        # Build and run (fastest)
